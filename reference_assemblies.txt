import pandas as pd
import re
import nltk
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import GaussianNB
from sklearn.metrics import confusion_matrix

nltk.download('stopwords')
dataset = [["I liked the movie", "positive"],
    ["It’s a good movie. Nice story", "positive"],
    ["Hero’s acting is bad but heroine looks good", "positive"],
    ["Overall nice movie", "positive"],
    ["Nice songs. But sadly boring ending.", "negative"],
    ["sad movie, boring movie", "negative"]]
df = pd.DataFrame(dataset,columns=['Text','Reviews'])
corpus = []
ps = PorterStemmer()
stop_words = set(stopwords.words('english'))
for text in df['Text']:
    text = re.sub('[^a-zA-Z]',' ',text).lower().split()
    text = [ps.stem(word) for word in text if word not in stop_words]
    corpus.append(' '.join(text))
cv = CountVectorizer()
x = cv.fit_transform(corpus).toarray()
y = df['Reviews'].values
print(corpus,x,y)

x_train, x_test, y_train, y_test = train_test_split(x,y,test_size=0.2)
model = GaussianNB()
model.fit(x_train, y_train)
y_pred = model.predict(x_test)
cm = confusion_matrix(y_test, y_pred)
print(cm)

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.model_selection import train_test_split
from sklearn.naive_bayes import MultinomialNB
from sklearn.metrics import accuracy_score, classification_report
documents = [
"I love this product, it is amazing",
"This is the worst experience ever",
"Absolutely fantastic service",
"I am very disappointed with this item",
"Great quality and fast delivery",
"Terrible product, not recommended"
]
labels = [1, 0, 1, 0, 1, 0]

vectorizer = TfidfVectorizer(stop_words='english')
X = vectorizer.fit_transform(documents)

X_train, X_test, y_train, y_test = train_test_split(X, labels, test_size=0.3, random_state=42)

model = MultinomialNB()
model.fit(X_train, y_train)
y_pred = model.predict(X_test)
print("Accuracy:", accuracy_score(y_test, y_pred))
print("\nClassification Report:\n", classification_report(y_test, y_pred))

test_sentence = ["The product is excellent and I love it"]
test_vector = vectorizer.transform(test_sentence)
prediction = model.predict(test_vector)
if prediction[0] == 1:
    print("Sentiment: Positive")
else:
    print("Sentiment: Negative")


### PRACTICAL-5 CYK(cocke-younger-kasami)
    
G={'S':[('A','B'),('B','C')],
   'A':[('B','A'),('a',)],
   'B':[('C','C'),('b',)],
   'C':[('A','B'),('a',)]}
w="baaba"
n=len(w)
T=[[set() for _ in range(n)] for _ in range(n)]
for i,c in enumerate(w):
    for X in G:
        if (c,) in G[X]: T[i][0].add(X)
for l in range(2,n+1):
    for i in range(n-l+1):
        for k in range(1,l):
            for X in G:
                for YZ in G[X]:
                    if len(YZ)==2 and YZ[0] in T[i][k-1] and YZ[1] in T[i+k][l-k-1]:
                        T[i][l-1].add(X)
for l in range(n,0,-1):
    print(" ".join("{" + ",".join(sorted(T[i][l-1])) + "}" if T[i][l-1] else "∅"
                   for i in range(n-l+1)))
print("ACCEPTED" if 'S' in T[0][n-1] else "REJECTED")


#### Practical 6 Maximum Likelihood Estimation(MLE)

from collections import Counter
corpus = ["i love NLP", 'i love Machine Learning', 'I hate NLP']

def preprocess(sentence):
    return sentence.lower().split()
tokenized_corpus = [preprocess(sent) for sent in corpus]

uc = Counter()
for s in tokenized_corpus:
    uc.update(s)
total_words = sum(uc.values())

up = {word: count/total_words for word, count in uc.items()}
print('Probabilities:')
for word, prob in up.items():
    print(f'{word}:{prob:.2f}')

bc = Counter()
for n in tokenized_corpus:
    for i in range(len(n)-1):
        bc[(n[i],n[i+1])] += 1

bp = {(w1,w2):bc[(w1,w2)]/uc[w1] for (w1,w2) in bc}
print('Bigram Probabilities:')
for word, prob in bp.items():
    print(f'{word}:{prob:.2f}')

def sentence_prob(sen):
    tokens = preprocess(sen)
    prob = 1
    for i in range(len(tokens)-1):
        bigram = (tokens[i],tokens[i+1])
        prob *= bp.get(bigram,0)
    return prob

test = 'i love NLP'
print('\n Sentence Probability:', sentence_prob(test))


# PRACTICAL-8 Multi-Word Expression (MWE)

corpus = [
    "I love machine learning",
    "Machine learning is fun",
    "I live in New York",
    "New York is a big city",
    "I like ice cream",
    "Ice cream is delicious"
]
def preprocess(sentence):
    return sentence.lower().split()
tokenized_corpus = [preprocess(sent) for sent in corpus]
from collections import Counter
unigram_counts = Counter()
bigram_counts = Counter()
for sentence in tokenized_corpus:
    unigram_counts.update(sentence)
    for i in range(len(sentence) - 1):
        bigram = (sentence[i], sentence[i+1])
        bigram_counts[bigram] += 1
threshold = 2  
mwe_list = []
for bigram, count in bigram_counts.items():
    if count >= threshold:
        mwe_list.append(" ".join(bigram))

print("Detected Multi-Word Expressions (MWEs):")
for mwe in mwe_list:
    print(mwe)
